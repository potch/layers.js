let avatar = new Layers.Stack({
  width: 400,
  height: 400,
  layers: [
    {
      name: 'mask',
      mode: 'destination-in',
      image: {
        url: "mask.png",
        size: "contain"
      }
    },
    {
      name: "shadows",
      mode: "screen",
      fill: "#004488"
    },
    {
      name: "highlights",
      mode: "multiply",
      fill: "#ffcc00"
    },
    {
      name: "curves",
      filter: Layers.filters.CurvesFilter,
      filterOptions: {
        amount: 0
      }
    },
    {
      filter: Layers.filters.GrayscaleFilter
    },
    {
      name: "image",
      image: {
        size: "cover"
      }
    },
    {
      fill: "#ffffff"
    }
  ]
});
